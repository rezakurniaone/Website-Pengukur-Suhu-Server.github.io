  hasil = 0
  for i in range(n):
    Li = 1.0
    for j ≠ i:
      Li *= (x - X[j]) / (X[i] - X[j])
    hasil += Y[i] * Li
  return hasil

function buatTabelSelisih(X, Y):
  tbl[i][0] = Y[i]  // orde-0
  tbl[i][j] = (tbl[i+1][j-1] - tbl[i][j-1])
              / (X[i+j] - X[i])

function hitungSuhuNewton(X, tbl, x):
  P = tbl[0][0]; prod = 1
  for i = 1 .. n:
    prod *= (x - X[i-1])
    P += tbl[0][i] * prod
  return P
